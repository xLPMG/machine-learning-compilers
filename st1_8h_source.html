64 constexpr uint32_t st1(simd_fp_t reg_dst,
65 gpr_t reg_src,
66 uint32_t index,
67 neon_size_spec_t size)
68 {
69 internal::checkSizeST1(size);
70 internal::checkIndexST1(size, index);
71
72 uint32_t l_ins = 0xD000000;
73 uint32_t l_opc = 0x4; // 100
74 uint32_t Q = 0x0;
75 uint32_t S = 0x0;
76 if (size == neon_size_spec_t::s)
77 {
78 S = index & 0x1;
79 Q = (index >> 1) & 0x1;
80 }
81 else if (size == neon_size_spec_t::d)
82 {
83 S = 0x0;
84 Q = index & 0x1;
85 }
86
87 l_ins |= (size << 10);
88 l_ins |= (l_opc << 13);
89 l_ins |= (Q << 30);
90 l_ins |= (S << 12);
91
92 uint32_t l_reg_id = reg_dst & 0x1f;
93 l_ins |= l_reg_id;
94
95 l_reg_id = reg_src & 0x1f;
96 l_ins |= l_reg_id << 5;
97
98 return l_ins;
99 }
